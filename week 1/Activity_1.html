numbers = []
def roll_d6():
  return random.randint(1, 6)

for _ in range(10):
    rollOutcome = roll_d6()
    numbers.append(rollOutcome)
print("roll outcomes: " + str(numbers))
se=pd.Series(numbers)

    'Size' : [se.size],
    'Mean': [se.mean()],
    'Median': [se.median()],
